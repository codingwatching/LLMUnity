169 void AssignLLM()
170 {
171 if (remote || llm != null) return;
172
173 LLM[] existingLLMs = FindObjectsOfType<LLM>();
174 if (existingLLMs.Length == 0) return;
175
176 SortBySceneAndHierarchy(existingLLMs);
177 llm = existingLLMs[0];
178 string msg = $"Assigning LLM {llm.name} to LLMCharacter {name}";
179 if (llm.gameObject.scene != gameObject.scene) msg += $" from scene {llm.gameObject.scene}";
180 LLMUnitySetup.Log(msg);
181 }

183 void SortBySceneAndHierarchy(LLM[] array)
184 {
185 for (int i = 0; i < array.Length - 1; i++)
186 {
187 bool swapped = false;
188 for (int j = 0; j < array.Length - i - 1; j++)
189 {
190 bool sameScene = array[j].gameObject.scene == array[j + 1].gameObject.scene;
191 bool swap = (
192 (!sameScene && array[j + 1].gameObject.scene == gameObject.scene) ||
193 (sameScene && array[j].transform.GetSiblingIndex() > array[j + 1].transform.GetSiblingIndex())
194 );
195 if (swap)
196 {
197 LLM temp = array[j];
198 array[j] = array[j + 1];
199 array[j + 1] = temp;
200 swapped = true;
201 }
202 }
203 if (!swapped) break;
204 }

478 public async Task<string> Chat(string query, Callback<string> callback = null, EmptyCallback completionCallback = null, bool addToHistory = true)
479 {
480 // handle a chat message by the user
481 // call the callback function while the answer is received
482 // call the completionCallback function when the answer is fully received
483 await LoadTemplate();
484 if (!CheckTemplate()) return null;
485 await InitNKeep();
486
487 string json;
488 await chatLock.WaitAsync();
489 try
490 {
491 AddPlayerMessage(query);
492 string prompt = template.ComputePrompt(chat, playerName, AIName);
493 json = JsonUtility.ToJson(GenerateRequest(prompt));
494 chat.RemoveAt(chat.Count - 1);
495 }
496 finally
497 {
498 chatLock.Release();
499 }
500
501 string result = await CompletionRequest(json, callback);
502
503 if (addToHistory && result != null)
504 {
505 await chatLock.WaitAsync();
506 try
507 {
508 AddPlayerMessage(query);
509 AddAIMessage(result);
510 }
511 finally
512 {
513 chatLock.Release();
514 }
515 if (save != "") _ = Save(save);
516 }
517
518 completionCallback?.Invoke();
519 return result;
520 }

708 protected async Task<Ret> PostRequestLocal<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
709 {
710 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
711 // this function has streaming functionality i.e. handles the answer while it is being received
712 string callResult = null;
713 bool callbackCalled = false;
714 while (!llm.failed && !llm.started) await Task.Yield();
715 switch (endpoint)
716 {
717 case "tokenize":
718 callResult = await llm.Tokenize(json);
719 break;
720 case "detokenize":
721 callResult = await llm.Detokenize(json);
722 break;
723 case "embeddings":
724 callResult = await llm.Embeddings(json);
725 break;
726 case "slots":
727 callResult = await llm.Slot(json);
728 break;
729 case "completion":
730 Callback<string> callbackString = null;
731 if (stream && callback != null)
732 {
733 if (typeof(Ret) == typeof(string))
734 {
735 callbackString = (strArg) =>
736 {
737 callback(ConvertContent(strArg, getContent));
738 };
739 }
740 else
741 {
742 LLMUnitySetup.LogError($"wrong callback type, should be string");
743 }
744 callbackCalled = true;
745 }
746 callResult = await llm.Completion(json, callbackString);
747 break;
748 default:
749 LLMUnitySetup.LogError($"Unknown endpoint {endpoint}");
750 break;
751 }
752
753 Ret result = ConvertContent(callResult, getContent);
754 if (!callbackCalled) callback?.Invoke(result);
755 return result;
756 }

758 protected async Task<Ret> PostRequestRemote<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
759 {
760 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
761 // this function has streaming functionality i.e. handles the answer while it is being received
762 if (endpoint == "slots")
763 {
764 LLMUnitySetup.LogError("Saving and loading is not currently supported in remote setting");
765 return default;
766 }
767
768 Ret result = default;
769 byte[] jsonToSend = new System.Text.UTF8Encoding().GetBytes(json);
770 UnityWebRequest request = null;
771 string error = null;
772 int tryNr = numRetries;
773
774 while (tryNr != 0)
775 {
776 using (request = UnityWebRequest.Put($"{host}:{port}/{endpoint}", jsonToSend))
777 {
778 WIPRequests.Add(request);
779
780 request.method = "POST";
781 if (requestHeaders != null)
782 {
783 for (int i = 0; i < requestHeaders.Count; i++)
784 request.SetRequestHeader(requestHeaders[i].Item1, requestHeaders[i].Item2);
785 }
786
787 // Start the request asynchronously
788 var asyncOperation = request.SendWebRequest();
789 float lastProgress = 0f;
790 // Continue updating progress until the request is completed
791 while (!asyncOperation.isDone)
792 {
793 float currentProgress = request.downloadProgress;
794 // Check if progress has changed
795 if (currentProgress != lastProgress && callback != null)
796 {
797 callback?.Invoke(ConvertContent(request.downloadHandler.text, getContent));
798 lastProgress = currentProgress;
799 }
800 // Wait for the next frame
801 await Task.Yield();
802 }
803 WIPRequests.Remove(request);
804 if (request.result == UnityWebRequest.Result.Success)
805 {
806 result = ConvertContent(request.downloadHandler.text, getContent);
807 error = null;
808 break;
809 }
810 else
811 {
812 result = default;
813 error = request.error;
814 }
815 }
816 tryNr--;
817 }
818
819 if (error != null) LLMUnitySetup.LogError(error);
820 callback?.Invoke(result);
821 return result;
822 }
