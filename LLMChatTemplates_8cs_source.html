173 public virtual string ComputePrompt(List<ChatMessage> messages, string playerName, string AIName, bool endWithPrefix = true)
174 {
175 string chatPrompt = PromptPrefix();
176 int start = 0;
177 if (messages[0].role == "system")
178 {
179 chatPrompt += RequestPrefix() + SystemPrefix() + messages[0].content + SystemSuffix();
180 start = 1;
181 }
182 for (int i = start; i < messages.Count; i += 2)
183 {
184 if (i > start || start == 0) chatPrompt += RequestPrefix();
185 chatPrompt += PlayerPrefix(messages[i].role) + PrefixMessageSeparator() + messages[i].content + RequestSuffix();
186 if (i < messages.Count - 1)
187 {
188 chatPrompt += AIPrefix(messages[i + 1].role) + PrefixMessageSeparator() + messages[i + 1].content + PairSuffix();
189 }
190 }
191 if (endWithPrefix) chatPrompt += AIPrefix(AIName);
192 return chatPrompt;
193 }
