206 public static void Reset()
207 {
208 if (!File.Exists(movedCache)) return;
209 List<StringPair> movedPairs = JsonUtility.FromJson<ListStringPair>(File.ReadAllText(movedCache)).pairs;
210 if (movedPairs == null) return;
211
212 bool refresh = false;
213 foreach (var pair in movedPairs)
214 {
215 if (pair.source == "") refresh |= DeletePath(pair.target);
216 else refresh |= MoveAction(pair.target, pair.source, false);
